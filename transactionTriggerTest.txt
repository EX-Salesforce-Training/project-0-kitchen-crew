@isTest
private class TransactionTriggerTest {

    
    @isTest private static void TestSelectServices(){
        
        

        Test.startTest();
         //Create a transaction then query Serivces to see if it is populated
         // start the scope of test
        CustomCustomer__c customer= new CustomCustomer__c();
		customer.Name='ellen';
        customer.Address__c='kinross ave';
        customer.Customer_LastName__c='Loustaunau';
        insert customer;

        Transaction__c transNew= new Transaction__c();
        transNew.Name='Basic Plan';
        transNew.Customer__c=customer.Id;
        transNew.Internet_PlanType__c='Basic';
        
        
  
        insert transNew;
      	Test.stopTest();
                
        //Perform Test to see if services is populated
        List<ServicesCustomer__c> listServices= [Select Name FROM ServicesCustomer__c Where Name= :transNew.Internet_PlanType__c];
        
        system.assertEquals(1,listServices.size());
    }
}
