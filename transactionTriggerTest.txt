public class TransactionTriggerHandler {

    

    
    public static void addRecords(List<Transaction__c> tre){
        
        
        //Insert new record in service customer 
        List<ServicesCustomer__c> custList= new List<ServicesCustomer__c>();
        List<ActiveWarrantys__c> warrantyList=new List<ActiveWarrantys__c>();
        List<Payment__c> paymentList=new List<Payment__c>();
        
        for(Transaction__c trans: tre){
        if(trans.TransactionType__c=='Internet'){
           
         ServicesCustomer__C customer= new ServicesCustomer__C();
            
        // customer.Name=trans.Internet_PlanType__c;
         customer.Name=trans.Name;
            
         custList.add(customer);
            
                insert custList;
        }
            
         if(trans.TransactionType__c=='Warranty'){
             
                ActiveWarrantys__C warranty=new ActiveWarrantys__C();
                
            	
               warranty.Name=trans.Name;
                warrantyList.add(warranty);
                insert warrantyList;
         }
            
            if(trans.TransactionType__c=='Payment'){
                Payment__c payments= new Payment__c();
                payments.payment_Amount__c=trans.PaymentAmount__c;
                payments.Name=trans.Name;
                paymentList.add(payments);
                 insert paymentList;
            }
         
        }
        
        
       
      
         
        
    }
}
