        Set<String> emailSet= new Set<String>();
        
        for(Contact c : newContacts){
            if(c.email != null){
                emailSet.add(c.email);
            }
        }
        
        for(Contact c: [Select email from contact where email in: emailSet]){
            for(Contact newCon: newContacts){

    }
